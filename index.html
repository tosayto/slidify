shinyUI(fluidPage(

  sidebarLayout(
    sidebarPanel(
      selectInput('x','Select your fact',names[-1])
    ),
    # Show a plot of the chosen fact
    mainPanel(
            h3(textOutput("factTitle")),
      htmlOutput("distPlot")
    )
  )
))
